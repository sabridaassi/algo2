tab2 := array_of integer[]

read(tab1[i])
end for
for j from 0 to n-1 do
read(tab2[j])
end for
sum=0
for i from 0 to tab1.length-1 do
i=0
   for j from 0 to tab2.length-1
       if tab1[i]<>tab2[j]
        sum:=sum+tab1[i]+tab2[j]
           i++
        end if
    end for
  end for
end
